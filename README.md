let state={level:"all",cat:"all",sess:[]};
const LVL_ORDER={beginner:0,intermediate:1,advanced:2};
const now=new Date();
const doy=Math.floor((now-new Date(now.getFullYear(),0,0))/86400000);
const q=QUOTES[doy%QUOTES.length];
document.getElementById("qt").textContent=`"${q.text}"`;
document.getElementById("qa").textContent=`— ${q.author}`;

function catLabel(id){return CATS.find(c=>c.id===id)?.label??id;}
function inSess(id){return state.sess.some(d=>d.id===id);}
function setLvl(l,btn){state.level=l;document.querySelectorAll(".lb").forEach(b=>b.classList.toggle("active",b===btn));render();}
function clearSess(){state.sess=[];render();}
function toggleSess(id){
  const drill=DRILLS.find(d=>d.id===id);
  if(!drill)return;
  state.sess=inSess(id)?state.sess.filter(d=>d.id!==id):[...state.sess,drill];
  render();
}

function getFiltered(){
  const sq=(document.getElementById("srch")?.value||"").trim().toLowerCase();
  const srt=document.getElementById("srt")?.value||"name";
  let list=DRILLS.filter(d=>{
    if(state.level!=="all"&&d.level!==state.level)return false;
    if(state.cat!=="all"&&d.category!==state.cat)return false;
    if(sq&&![d.name,d.summary,d.category,d.level,...d.equipment].join(" ").toLowerCase().includes(sq))return false;
    return true;
  });
  list.sort((a,b)=>{
    if(srt==="duration")return a.duration-b.duration;
    if(srt==="level")return(LVL_ORDER[a.level]??9)-(LVL_ORDER[b.level]??9);
    return a.name.localeCompare(b.name);
  });
  return list;
}

function renderCats(){
  document.getElementById("catlist").innerHTML=CATS.map(c=>{
    const cnt=DRILLS.filter(d=>{
      if(state.level!=="all"&&d.level!==state.level)return false;
      return c.id==="all"||d.category===c.id;
    }).length;
    return `<li class="ci ${state.cat===c.id?"active":""}" onclick="state.cat='${c.id}';render()">
      <span>${c.icon} ${c.label}</span><span class="cc">${cnt}</span>
    </li>`;
  }).join("");
}

function renderSess(){
  const n=state.sess.length,tot=state.sess.reduce((s,d)=>s+d.duration,0);
  document.getElementById("scnt").textContent=n===0?"0 drills selected":`${n} drill${n>1?"s":""} · ~${tot} min`;
  document.getElementById("slist").innerHTML=state.sess.map(d=>
    `<div class="sdrill"><span>${d.name}</span><span style="cursor:pointer;color:var(--acc);font-size:.9rem" onclick="toggleSess('${d.id}')">×</span></div>`
  ).join("");
}

function renderStats(f){
  const b=f.filter(d=>d.level==="beginner").length,i=f.filter(d=>d.level==="intermediate").length,a=f.filter(d=>d.level==="advanced").length,m=f.reduce((s,d)=>s+d.duration,0);
  document.getElementById("stats").innerHTML=`<span class="chip"><strong>${f.length}</strong> drills</span><span class="chip"><strong>${b}</strong> beginner</span><span class="chip"><strong>${i}</strong> intermediate</span><span class="chip"><strong>${a}</strong> advanced</span><span class="chip"><strong>${m}</strong> min total</span>`;
}

  const f=getFiltered();
  renderStats(f);
  const g=document.getElementById("grid");

  g.innerHTML=f.map(d=>`
    <div class="dc" onclick="openModal('${d.id}')">
      <div class="dch"><h3>${d.name}</h3><span class="badge ${d.level}">${d.level}</span></div>

    </div>`).join("");
}
